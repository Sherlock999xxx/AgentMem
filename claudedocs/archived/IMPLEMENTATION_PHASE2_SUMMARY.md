#### 配置选项
```rust
pub struct OrchestratorConfig {
    pub enable_embedder_cache: Option<bool>,        // 默认: true
    pub embedder_cache_size: Option<usize>,         // 默认: 1000
    pub embedder_cache_ttl_secs: Option<u64>,       // 默认: 3600
}
```

#### 使用示例
```rust
// 默认配置 (推荐)
let memory = Memory::new_core().await?;

// 自定义配置
let config = OrchestratorConfig {
    enable_embedder_cache: Some(true),
    embedder_cache_size: Some(2000),
    embedder_cache_ttl_secs: Some(7200),
    ..Default::default()
};
let memory = Memory::new_with_config(config).await?;
```
